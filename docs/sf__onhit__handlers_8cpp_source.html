71{
72 // no need to calculate damage in this cases

92 spellAPI.getResourceSpellData(_this->CGdResource, &spell_data, spell_id);
93
94 uint16_t tmp_damage = damage * spell_data.params[1];
95 uint16_t hp_left = figureAPI.getCurrentHealth(_this->CGdFigure, target_index);
96 if (tmp_damage < hp_left)
97 {
98 return damage;
99 }
100
101 tmp_damage = ((tmp_damage - hp_left) * 100) / tmp_damage;
102 uint16_t chance = 5 * tmp_damage * (5 + (int16_t)(source_level - target_level));
103 if (spellAPI.getRandom(_this->OpaqueClass, 10000) > chance)
104 {
105 return damage;
106 }
