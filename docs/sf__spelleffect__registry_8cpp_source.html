12void registerEffectHandler(uint16_t spell_job, handler_ptr handler)
13{
14 auto check = s_effect_handler_map.find(spell_job);
15 if (check != s_effect_handler_map.end())
16 {
17 char message[256];
18 sprintf(message, "%s (v%s) has replaced an Effect Handler [%d] (Was this on purpose?)", g_current_mod->mod_id, g_current_mod->mod_version, spell_job);
19 log_warning(message);
20 }
21 s_effect_handler_map[spell_job] = handler;
22}

24handler_ptr get_spell_effect(uint16_t spell_job)
25{
26 auto it = s_effect_handler_map.find(spell_job);
27 if (it == s_effect_handler_map.end())
28 {
29 char message[256];
30 sprintf(message, "%s Unknown Job ID [%d], No effect handler registered.", g_current_mod->mod_id, spell_job);
31 log_warning(message);

34 return it->second;
